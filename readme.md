# Getting started

1- composer install

2- set .env variables

3- create your database

4- run command php artisan migrate

5- insert into table cfg_status {
    STATUS_ACTIVE = 1;
    STATUS_SUSPENDED = 2;
    STATUS_BLOCK = 3;
    STATUS_PENDING = 4;
}

6- insert into table cfg_domain_status {
    STATUS_ACTIVE = 1;
    STATUS_SUSPENDED = 2;
    STATUS_BLOCK = 3;
    STATUS_PENDING = 4;
}
